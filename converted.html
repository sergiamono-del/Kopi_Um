osc2.type = 'triangle';

osc2.frequency.value = 220;

gain2.gain.value = 0.05;

osc2.connect(gain2).connect(master);

osc2.start();

const lfo = audioCtx.createOscillator();

const lfoGain = audioCtx.createGain();

lfo.type = 'sine';

lfo.frequency.value = 0.08;

lfoGain.gain.value = 0.012;

lfo.connect(lfoGain).connect(master.gain);

lfo.start();

nodes = { osc1, osc2, lfo, master };

}

function stopAmbience() {

if (!nodes || !audioCtx) return;

const now = audioCtx.currentTime;

nodes.master.gain.cancelScheduledValues(now);

nodes.master.gain.setTargetAtTime(0, now, 0.08);

setTimeout(() => {

try { nodes.osc1.stop(); } catch(e) {}

try { nodes.osc2.stop(); } catch(e) {}

try { nodes.lfo.stop(); } catch(e) {}

try { nodes.osc1.disconnect(); nodes.osc2.disconnect(); nodes.lfo.disconnect(); nodes.master.disconnect(); } catch(e) {}

nodes = null;

}, 180);

}

musicBtn.addEventListener('click', () => {

musicOn = !musicOn;

if (musicOn) {

startAmbience();

musicBtn.textContent = '⏸ Pause Suara';

musicState.textContent = 'Suara aktif';

} else {

stopAmbience();

musicBtn.textContent = '▶ Putar Suara';

musicState.textContent = 'Suara mati';

}

});

const gameStatus = document.getElementById('gameStatus');

const startGameBtn = document.getElementById('startGame');

let score = 0;

let timeLeft = 30;

let gameTimer = null;

let moveTimer = null;

let gameRunning = false;

function moveBean() {

const rect = gameStage.getBoundingClientRect();

const maxX = Math.max(20, rect.width - 60);

const maxY = Math.max(20, rect.height - 50);

const x = Math.random() * maxX;

const y = Math.random() * maxY;

bean.style.left = `${x}px`;

bean.style.top = `${y}px`;

}

function spark(x, y) {

const dot = document.createElement('div');

dot.className = 'bean-spark';

dot.style.left = `${x}px`;

dot.style.top = `${y}px`;

gameStage.appendChild(dot);

setTimeout(() => dot.remove(), 650);

}

function startGame() {

if (gameRunning) return;

gameRunning = true;

score = 0;

timeLeft = 30;

scoreEl.textContent = score;

timeEl.textContent = timeLeft;

gameStatus.textContent = 'Berlangsung';

moveBean();

gameTimer = setInterval(() => {

timeLeft--;

timeEl.textContent = timeLeft;

if (timeLeft <= 0) {

clearInterval(gameTimer);

clearInterval(moveTimer);

gameRunning = false;

gameStatus.textContent = `Selesai — skor ${score}`;

bean.style.left = '50%';

bean.style.top = '50%';

bean.style.transform = 'translate(-50%, -50%) scale(1)';

}

}, 1000);

moveTimer = setInterval(() => {

if (gameRunning) moveBean();

}, 950);

}

function resetGame() {

clearInterval(gameTimer);

clearInterval(moveTimer);

score = 0;

timeLeft = 30;

gameRunning = false;

scoreEl.textContent = score;

timeEl.textContent = timeLeft;

gameStatus.textContent = 'Siap dimainkan';

bean.style.left = '24px';

bean.style.top = '90px';

bean.style.transform = 'none';

}

bean.addEventListener('click', (e) => {

if (!gameRunning) return;

score++;

scoreEl.textContent = score;

const rect = gameStage.getBoundingClientRect();

spark(e.offsetX + bean.offsetLeft, e.offsetY + bean.offsetTop);

moveBean();

});

startGameBtn.addEventListener('click', startGame);

resetGameBtn.addEventListener('click', resetGame);
